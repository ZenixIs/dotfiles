INC		=	-I./include/

CC		=	gcc $(INC)

SRCDIR	=	src

SRC		=   $(SRCDIR)/main.c

OBJ		=	$(SRC:.c=.o)

BIN		=	bin

CFLAGS	=	-W -Wall -Wextra

all: $(OBJ)
	$(CC) $(CFLAGS) -o $(BIN) $(OBJ) $(LIB)

clean:
		rm -f $(BIN)

fclean:	clean
		rm -f $(OBJ)

re:		fclean all

.PHONY:		all clean fclean re
